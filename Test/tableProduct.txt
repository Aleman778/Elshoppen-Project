CREATE TABLE PRODUCT (
	id INT(30) UNSIGNED AUTO_INCREMENT NOT NULL, 
	product_name VARCHAR(100) NOT NULL,
	price INT(30) UNSIGNED AUTO_INCREMENT NOT NULL,
	inventory INT(1000) UNSIGNED AUTO_INCREMENT NOT NULL,
	image VARCHAR(1000) NOT NULL,
	category VARCHAR(1000) NOT NULL,
	description VARCHAR(1000) NOT NULL
);

CREATE TABLE SHOPPING_CART (
	customer_id INT(30) UNSIGNED AUTO_INCREMENT NOT NULL,
	product_id INT(30) UNSIGNED AUTO_INCREMENT NOT NULL,
	quantity INT(100) UNSIGNED AUTO_INCREMENT NOT NULL
);
